local sprotoparser = require "sprotoparser"

local proto = {}

proto.c2s = sprotoparser.parse [[
.package {
	type 0 : integer
	session 1 : integer
}

hello 1 {
	response {
		msg 0  : string
	}
}

heartbeat 2 {
	request {
		what 0 : string
	}
	response {
		result 0 : string
	}
}

quit 3 {}

]]

proto.s2c = sprotoparser.parse [[
.package {
	type 0 : integer
	session 1 : integer
}

set 1 {
	request {
		key 0 : string
		value 1 : string
	}
}

]]

return proto
