--[[
登录服管理：
每个服务器唯一
监视所有登录服
将登录服踢出
接收登录服的加入
--]]
local skynet = require("skynet")
local snax = require "skynet.snax"

local loginservices = {}

local function getloginservice(index)
    return loginservices[index]
end

local function newloginservice()
    local s = skynet.newservice("loginservice")
    loginservices[#loginservices + 1] = s
    return #loginservices
end

function init(...)
    print("login service manager init:", ...)

    --创建loginservice规则待定
    --这里先创建
    newloginservice()
end

function exit(...)
    print("login service manager exit:", ...)
end
